module EquivalenceLemmas where

  equiv-eq : ∀ {i} {A B : Set i} {f g : A ≃ B} → π₁ f ≡ π₁ g → f ≡ g
  equiv-eq p = Σ-eq p $ prop-has-all-paths (is-equiv-is-prop _) _ _
